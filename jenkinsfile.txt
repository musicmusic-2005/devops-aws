pipeline {
    agent any

    stages {
        stage('Deploy to S3') {
            steps {
                withCredentials([[
                    $class: 'AmazonWebServicesCredentialsBinding',
                    credentialsId: 'aws-s3-creds'
                ]]) {
                    sh '''
                    aws s3 sync . s3://edwin-static-site --delete
                    '''
                }
            }
        }
    }
}
